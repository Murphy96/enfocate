# Listas Dobles Circular

Clase **LD_Circular**:
```java

public class LD_Circular {
	private NodoDoble p;
	public LD_Circular() {
		p = null; 
	}
	public boolean esVacia() {
		return p == null ? true : false;
	}
	public void adiUltimo(int dato){
		NodoDoble nuevo = new NodoDoble();
		nuevo.setDato(dato);
		if(esVacia()){
			p = nuevo;
			p.setAnt(nuevo);
			nuevo.setSig(p);
		}else{
			p.getAnt().setSig(nuevo);
			nuevo.setSig(p);
			nuevo.setAnt(p.getAnt());
			p.setAnt(nuevo);
		}
	}
	public void mostrar(){
		NodoDoble r = p;
		while(r.getSig()!= p){
			System.out.print("<- " + r.getDato()+ " ->");
			r = r.getSig();
		}
		System.out.println("<- " + r.getDato()+ " ->");
	}
}

```
Clase **NodoDoble**:
```java
public class NodoDoble {
	private int dato;
	private NodoDoble ant;
	private NodoDoble sig;

	public NodoDoble getAnt() {
		return ant;
	}

	public void setAnt(NodoDoble ant) {
		this.ant = ant;
	}

	public NodoDoble() {
		dato = 0;
		sig = null;
		ant = null;
	}

	public int getDato() {
		return dato;
	}

	public void setDato(int dato) {
		this.dato = dato;
	}

	public NodoDoble getSig() {
		return sig;
	}

	public void setSig(NodoDoble sig) {
		this.sig = sig;
	}
}


```
Clase **Main**:
```java
import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
		int n = 3;
		Scanner sc=new Scanner(System.in);
		LD_Circular lista= new LD_Circular();
		for (int i = 0; i < n; i++) {
			System.out.println("Intrese un dato a la lista <int>");
			int nuevoDato = sc.nextInt();
			lista.adiUltimo(nuevoDato);
		}
		lista.mostrar();
	}
}
```
