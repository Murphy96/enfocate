# Listas Dobles Normal


Clase NodoDoble
```java
public class NodoDoble {
	private int dato;
	private NodoDoble ant;
	private NodoDoble sig;

	public NodoDoble getAnt() {
		return ant;
	}

	public void setAnt(NodoDoble ant) {
		this.ant = ant;
	}

	public NodoDoble() {
		dato = 0;
		sig = null;
		ant = null;
	}

	public int getDato() {
		return dato;
	}

	public void setDato(int dato) {
		this.dato = dato;
	}

	public NodoDoble getSig() {
		return sig;
	}

	public void setSig(NodoDoble sig) {
		this.sig = sig;
	}
}
```

Clase Lista
```java
public class LD_Normal {
	private NodoDoble p;

	public LD_Normal() {
		p = null;
	}

	public boolean esVacia() {
		return p == null ? true : false;
	}

	public void adiUltimo(int dato) {
		NodoDoble nuevoNodo = new NodoDoble();
		nuevoNodo.setDato(dato);
		if (esVacia())
			p = nuevoNodo;
		else {
			NodoDoble r = p;
			while (r.getSig() != null)
				r = r.getSig();
			r.setSig(nuevoNodo);
			nuevoNodo.setAnt(r);
		}
	}
	public void adiPrimero(int dato){
		NodoDoble nuevoNodo = new NodoDoble();
		nuevoNodo.setDato(dato);
		if (esVacia())
			p = nuevoNodo;
		else{
			p.setAnt(nuevoNodo);
			nuevoNodo.setSig(p);
			p = nuevoNodo;
		}
	}
	public void mostrar() {
		NodoDoble r = p;
		while (r.getSig() != null) {
			System.out.print("<- " + r.getDato() + " ->");
			r = r.getSig();
		}
		System.out.print("<- " + r.getDato() + " -> null");
	}
}

```

import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n = 3;
		LD_Normal ldn = new LD_Normal();
		for (int i = 0; i < n; i++) {
			ldn.adiPrimero(sc.nextInt());
		}
		ldn.mostrar();
	}
}
